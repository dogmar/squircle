Animate Component Model

<Animate inStep outStep> controls visibility lifecycle

The Animate Component

type AnimateProps = {
  inStep: number; // Appears at this step
  outStep?: number; // Disappears at this step (omit = stays)
  children: React.ReactNode;
  // Optional Framer Motion overrides:
  initial?: MotionProps; // default: { opacity: 0, y: 8 }
  animate?: MotionProps; // default: { opacity: 1, y: 0 }
  exit?: MotionProps; // default: { opacity: 0, y: -8 }
};

// Visible when: inStep <= currentStep < outStep
function Animate({ inStep, outStep, children, ...motion }) {
  const step = useSlideStep();
  const visible = step >= inStep && (outStep == null || step < outStep);

  return (
    <AnimatePresence>
      {visible && (
        <motion.div key={inStep}
          initial={motion.initial ?? { opacity: 0, y: 8 }}
          animate={motion.animate ?? { opacity: 1, y: 0 }}
          exit={motion.exit ?? { opacity: 0, y: -8 }}
        />
      )}
    </AnimatePresence>
  );
}

Example Slide

{
  content: (
    <>
      <h2>The Rounded Corner</h2>

      {/* Appears at step 0, stays forever */}
      <Animate inStep={0}>
        <p>CSS border-radius draws a circular arc.</p>
      </Animate>

      {/* Appears at step 1, exits at step 3 */}
      <Animate inStep={1} outStep={3}>
        <p>Notice the abrupt transition here.</p>
      </Animate>

      {/* Appears at step 2, custom animation */}
      <Animate inStep={2}
        initial={{ opacity: 0, scale: 0.95 }}
        animate={{ opacity: 1, scale: 1 }}
      >
        <div className="callout">
          This is because arcs are circular segments.
        </div>
      </Animate>

      {/* Replaces the step-1 text at step 3 */}
      <Animate inStep={3}>
        <p>Can we do better? Enter the superellipse.</p>
      </Animate>
    </>
  ),

  steps: [
    { graphic: { showRounded: true } },
    { graphic: { showMeasurement: true } },
    { /* text only — no graphic change */ },
    { graphic: { showSuperellipse: true } },
  ],
}
